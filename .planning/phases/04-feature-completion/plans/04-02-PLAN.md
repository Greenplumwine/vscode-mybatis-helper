export class PropertyStrategy extends BaseCompletionStrategy {
  readonly triggerCharacters = ['.'] as const;
  readonly priority = 80;
  readonly name = 'Property';

  canComplete(context: CompletionContext): boolean {
    const pattern = /#\{(\w+)\.[\w]*$/;
    const altPattern = /\$\{(\w+)\.[\w]*$/;
    return pattern.test(context.linePrefix) || altPattern.test(context.linePrefix);
  }

  async provideCompletionItems(context: CompletionContext): Promise<vscode.CompletionItem[]>
}

<risk_areas>
1. **递归风险：** 循环引用类型可能导致无限递归（A 有 B 属性，B 有 A 属性），需要 visitedTypes 检测
2. **性能风险：** 多级属性解析可能触发多次文件解析，大项目需要限制深度
3. **并发风险：** 相同类型的并发解析请求需要合并，避免重复文件读取
4. **类型解析风险：** 泛型类型擦除可能导致类型信息不准确（List<User> 解析为 List）
5. **文件 IO 风险：** 频繁的源码文件读取可能影响性能，需要 LRU 缓存
</risk_areas>
