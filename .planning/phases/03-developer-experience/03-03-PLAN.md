     configPath: string;
     severity: 'error' | 'warning' | 'info';

2. src/services/validation/configurationValidator.ts:
   Export functions:
   - `validateConfiguration(): Promise<ValidationResult>` - Full validation
   - `validateBasic(): ValidationIssue[]` - Quick validation

   Validation checks:

   a) customXmlDirectories:
      - Check each path exists (fs.existsSync)
      - Check path is within workspace (security)
      - Severity: error if not exists, warning if outside workspace

   b) nameMatchingRules:
      - Check each pattern is valid regex
      - Try new RegExp(pattern) and catch errors
      - Severity: error if invalid regex

   c) sqlInterceptor.customRules:
      - Check each regex pattern is valid
      - Severity: error if invalid regex

   d) sqlInterceptor.listenMode:
      - Check value is one of: auto, debugConsole, terminal
      - Severity: error if invalid

   e) databaseType:
      - Check value is valid dialect
      - Severity: warning if unknown (may still work)

   f) pathPriority:
      - Check each path pattern is valid
      - Severity: warning if looks suspicious

   For each issue, provide:
   - configPath: dot-notation path (e.g., "mybatis-helper.customXmlDirectories[0]")
   - severity: error/warning/info
   - message: Clear description of the problem
   - suggestion: Actionable fix (e.g., "Create directory or remove from configuration")

   const configWatcher = vscode.workspace.onDidChangeConfiguration(e => {

     }
