## Task 2: Add Java method parser tests

<read_first>
- src/services/language/javaMethodParser.ts
- src/services/language/javaMethodParser.d.ts (if exists)
</read_first>

<action>
Create comprehensive tests for Java method parser:

1. Create `src/services/language/__tests__/javaMethodParser.test.ts`:
   ```typescript
   import { parseJavaMethods, JavaMethodInfo } from '../javaMethodParser';

   describe('javaMethodParser', () => {
     describe('parseJavaMethods', () => {
       it('should parse simple method', () => {
         const content = `
           public interface UserMapper {
             User selectById(Long id);
           }
         `;
         const methods = parseJavaMethods(content, 'UserMapper.java');
         expect(methods).toHaveLength(1);
         expect(methods[0].name).toBe('selectById');
         expect(methods[0].returnType).toBe('User');
         expect(methods[0].parameters).toHaveLength(1);
         expect(methods[0].parameters[0].type).toBe('Long');
         expect(methods[0].parameters[0].name).toBe('id');
       });

       it('should parse method with multiple parameters', () => {
         const content = `
           public interface UserMapper {
             List<User> selectByCondition(String name, Integer age, Date createTime);
           }
         `;
         const methods = parseJavaMethods(content, 'UserMapper.java');
         expect(methods).toHaveLength(1);
         expect(methods[0].parameters).toHaveLength(3);
       });

       it('should parse method with annotations', () => {
         const content = `
           public interface UserMapper {
             @Select("SELECT * FROM users WHERE id = #{id}")
             User selectById(Long id);
           }
         `;
         const methods = parseJavaMethods(content, 'UserMapper.java');
         expect(methods).toHaveLength(1);
         expect(methods[0].annotations).toContain('@Select');
       });

       it('should handle generic return types', () => {
         const content = `
           public interface UserMapper {
             List<User> selectAll();
             Map<String, Object> selectAsMap();
           }
         `;
         const methods = parseJavaMethods(content, 'UserMapper.java');
         expect(methods).toHaveLength(2);
         expect(methods[0].returnType).toBe('List<User>');
         expect(methods[1].returnType).toBe('Map<String, Object>');
       });

       it('should parse void methods', () => {
         const content = `
           public interface UserMapper {
             void insert(User user);
             void updateById(User user);
             void deleteById(Long id);
           }
         `;
         const methods = parseJavaMethods(content, 'UserMapper.java');
         expect(methods).toHaveLength(3);
         methods.forEach(m => expect(m.returnType).toBe('void'));
       });

       it('should handle interface with @Mapper annotation', () => {
         const content = `
           @Mapper
           public interface UserMapper {
             User selectById(Long id);
           }
         `;
         const methods = parseJavaMethods(content, 'UserMapper.java');
         expect(methods).toHaveLength(1);
       });

       it('should return empty array for non-interface file', () => {
         const content = `
           public class UserService {
             public User getUser() { return null; }
           }
         `;
         const methods = parseJavaMethods(content, 'UserService.java');
         expect(methods).toHaveLength(0);
       });

       it('should handle complex generic parameters', () => {
         const content = `
           public interface UserMapper {
             List<User> selectByIds(List<Long> ids);
             Map<String, List<User>> selectGrouped();
           }
         `;
         const methods = parseJavaMethods(content, 'UserMapper.java');
         expect(methods).toHaveLength(2);
       });
     });
   });
   ```
</action>

<read_first>

</read_first>

<action>
Create tests for XML parsing:

1. Create `src/features/mapping/__tests__/xmlParser.test.ts`:
   ```typescript
   import { parseXmlMapper, extractNamespace, extractMethods } from '../xmlParser';

   describe('xmlParser', () => {
     const sampleMapper = `<?xml version="1.0" encoding="UTF-8"?>
       <!DOCTYPE mapper PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN" "http://mybatis.org/dtd/mybatis-3-mapper.dtd">
       <mapper namespace="com.example.UserMapper">
         <select id="selectById" resultType="User">
           SELECT * FROM users WHERE id = #{id}
         </select>

         <insert id="insert">
           INSERT INTO users (name, email) VALUES (#{name}, #{email})
         </insert>

         <update id="updateById">
           UPDATE users SET name = #{name} WHERE id = #{id}
         </update>

         <delete id="deleteById">
           DELETE FROM users WHERE id = #{id}
         </delete>
       </mapper>`;

     describe('parseXmlMapper', () => {
       it('should parse namespace', () => {
         const result = parseXmlMapper(sampleMapper);
         expect(result.namespace).toBe('com.example.UserMapper');
       });

       it('should extract all method IDs', () => {
         const result = parseXmlMapper(sampleMapper);
         expect(result.methods).toHaveLength(4);
         expect(result.methods.map(m => m.id)).toEqual([
           'selectById', 'insert', 'updateById', 'deleteById'
         ]);
       });

       it('should identify statement types', () => {
         const result = parseXmlMapper(sampleMapper);
         expect(result.methods[0].type).toBe('select');
         expect(result.methods[1].type).toBe('insert');
         expect(result.methods[2].type).toBe('update');
         expect(result.methods[3].type).toBe('delete');
       });

       it('should extract resultType', () => {
         const result = parseXmlMapper(sampleMapper);
         expect(result.methods[0].resultType).toBe('User');
       });

       it('should handle empty mapper', () => {
         const emptyMapper = `<mapper namespace="com.example.EmptyMapper"></mapper>`;
         const result = parseXmlMapper(emptyMapper);
         expect(result.namespace).toBe('com.example.EmptyMapper');
         expect(result.methods).toHaveLength(0);
       });

       it('should parse SQL with CDATA', () => {
         const mapperWithCData = `<mapper namespace="TestMapper">
           <select id="complexQuery">
             <![CDATA[
               SELECT * FROM users WHERE age > #{minAge} AND age < #{maxAge}
             ]]>
           </select>
         </mapper>`;
         const result = parseXmlMapper(mapperWithCData);
         expect(result.methods).toHaveLength(1);
         expect(result.methods[0].sql).toContain('SELECT * FROM users');
       });
     });

     describe('extractNamespace', () => {
       it('should extract namespace from valid mapper', () => {
         const ns = extractNamespace(sampleMapper);
         expect(ns).toBe('com.example.UserMapper');
       });

       it('should return null for invalid XML', () => {
         const ns = extractNamespace('not xml');
         expect(ns).toBeNull();
       });
     });
   });
   ```
</action>

## Task 5: Create CI test workflow

<read_first>
- .github/workflows/ (if exists)
- package.json scripts
</read_first>

<action>
Create GitHub Actions workflow for automated testing:

1. Create `.github/workflows/test.yml`:
   ```yaml
   name: Tests

   on:
     push:
       branches: [main, master]
     pull_request:
       branches: [main, master]

   jobs:
     test:
       runs-on: ubuntu-latest

       strategy:
         matrix:
           node-version: [18.x, 20.x]

       steps:
         - uses: actions/checkout@v4

         - name: Setup Node.js ${{ matrix.node-version }}
           uses: actions/setup-node@v4
           with:
             node-version: ${{ matrix.node-version }}
             cache: 'pnpm'

         - name: Install pnpm
           uses: pnpm/action-setup@v2
           with:
             version: 8

         - name: Install dependencies
           run: pnpm install

         - name: Compile TypeScript
           run: pnpm run compile

         - name: Run linting
           run: pnpm run lint

         - name: Run unit tests
           run: pnpm run test:unit -- --coverage

         - name: Upload coverage
           uses: codecov/codecov-action@v3
           with:
             files: ./coverage/lcov.info
             flags: unittests
             name: codecov-umbrella
   ```
</action>

<acceptance_criteria>
- `.github/workflows/test.yml` exists
- Workflow triggers on push/PR to main/master
- Runs on Node 18.x and 20.x
- Runs compile, lint, and unit tests
